class Solution {
    public boolean isSubsequence(String s, String t) {
        int left = 0;
        int right = t.length()-1;
        int i=0;

        while(left<=right && i<s.length()) {
            if(s.charAt(i)==t.charAt(left)) {
                left++;
                i++;
            } else {
                left++;
            }
        }

        if(i>=s.length()) return true;

        return false;
    }
}
