class Solution {
    public int romanToInt(String s) {
        // Brute Force Approach
        int res = 0;
        int n = s.length();
        char[] romans = s.toCharArray();

        for(int i=0; i<n; i++) {
            if(romans[i]=='C' && i+1 < n && romans[i+1]=='M') {
                res += 900;
                i++;
            } else if(romans[i]=='C' && i+1 < n && romans[i+1]=='D') {
                res += 400;
                i++;
            } else if(romans[i]=='X' && i+1 < n && romans[i+1]=='C') {
                res += 90;
                i++;
            } else if(romans[i]=='X' && i+1 < n && romans[i+1]=='L') {
                res += 40;
                i++;
            } else if(romans[i]=='I' && i+1 < n && romans[i+1]=='X') {
                res += 9;
                i++;
            } else if(romans[i]=='I' && i+1 < n && romans[i+1]=='V') {
                res += 4;
                i++;
            } else if(romans[i]=='M') {
                res += 1000;
            } else if(romans[i]=='D') {
                res += 500;
            } else if(romans[i]=='C') {
                res += 100;
            } else if(romans[i]=='L') {
                res += 50;
            } else if(romans[i]=='X') {
                res += 10;
            } else if(romans[i]=='V') {
                res += 5;
            } else if(romans[i]=='I') {
                res += 1;
            }
        }

        return res;
    }
}
