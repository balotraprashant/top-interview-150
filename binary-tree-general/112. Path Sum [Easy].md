/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        int sum = 0;
        return hasTargetSum(root, targetSum, sum);
    }

    public boolean hasTargetSum(TreeNode root, int targetSum, int sum) {
        if(root==null) return false;

        sum = root.val+sum;

        if(sum == targetSum && (root.left==null || root.left.val==0) && (root.right==null || root.right.val==0)) return true;

        else {
            return hasTargetSum(root.left, targetSum, sum) || hasTargetSum(root.right, targetSum, sum);
        }
    }
}
